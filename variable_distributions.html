To assess whether the visual appeal of the cues influences participants’ ratings.

load([data_folder      'post_task_qs.mat']);
qdata = pX_restructure_post_task_qs(FINAL_Q_DATA);
qdata = outerjoin(qdata, dt_used, 'Keys','specID');
qdata(isnan(qdata.c1_appeal),:) =[];
qdata.contingency = qdata.contingency_dt_used;
qdata.specID = qdata.specID_dt_used;

xtickcstr = {'Stable-high', 'Stable-low', 'Reversal'};
cf=struct;
cf.opacity =0.5;
cf.legendmark_2 = 'on';
vars  = {'appeal'};
cstr = strcat('c', strsplit(num2str(1:3), ' '));
for v = 1:numel(vars)
    nqdata = stack(qdata, {strcat('c', strsplit(num2str(1:3), ' '), '_',  vars{v})}, 'NewDataVariableName', vars{v}, 'IndexVariableName', 'cue');
    nqdata.cue = strrep(cellstr(nqdata.cue), ['_' vars{v}], '');
    
    f=figure;
    for b = 1:numel(blocks)
        subplot(1,3,b);
        for c = 1:numel(cstr)
            id = find( ismember(nqdata.cue, cstr{c}) & ismember(nqdata.contingency, blocks{b})  );
            plot_vertical_jittered_column(c, nqdata.(vars{v})(id), [0.3 0.6 0.4], cf)
        end

    end

    cf= []; 
    cf.bycat=1;
    cf.cat ={'contingency'};
    [m] = pX_report_stats(nqdata, vars{v}, { 'cue', 'contingency'}, {'ids', 'study'}, cf);
